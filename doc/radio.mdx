## 基础用法
<Playground>
  {() => {
    const [checked, setChecked] = React.useState('male')
    const [fruit, setFruit] = React.useState('apple')
    return (
      <div>
          <div>
              <Radio.Group value={checked} onChange={e => setChecked(e.target.value)}>
                <Radio value="male">男</Radio>
                <Radio value="female">女</Radio>
              </Radio.Group>
          </div>
          <div>
             <Radio.Group onChange={e => setFruit(e.target.value)} value={fruit}>
                <Radio value="apple">苹果</Radio>
                <Radio value="pears">梨子</Radio>
                <Radio value="cucumber" disabled>黄瓜</Radio>
              </Radio.Group>
          </div>
      </div>
      )
  }}
</Playground>

## 支持自定义比较函数
<Playground>
  {() => {
    const [checked, setChecked] = React.useState({ foo: 1 })
    const isValueEqual = (a, b) => {
      return a && b && a.foo === b.foo;
    }
    return (
      <div>
         <Radio.Group 
          value={checked} 
          onChange={e => setChecked(e.target.value)}
          isValueEqual={isValueEqual}
          >
            <Radio value={{ foo: 1 }}>Foo 1</Radio>
            <Radio value={{ foo: 2 }}>Foo 2</Radio>
          </Radio.Group>
      </div>
      )
  }}
</Playground>
