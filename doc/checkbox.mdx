## checkbox全选
<Playground>
{() => {
    const ITEMS = ['Item 1', 'Item 2', 'Item 3'];

    const [checkedAllList, setAllCheckedList] = React.useState([])

    const checkedAll =
      !!checkedAllList.length && checkedAllList.length === ITEMS.length;
    const indeterminate =
      !!checkedAllList.length && checkedAllList.length !== ITEMS.length;

    return (<div>
        <Checkbox
          checked={checkedAll}
          indeterminate={indeterminate}
          onChange={(e)=>setAllCheckedList(e.target.checked ? ITEMS.slice() : [])}
        >
          全选
        </Checkbox>
        <div style={{ margin: '5px 0', height: 1, background: '#dcdee0' }}></div>

        <Checkbox.Group
          value={checkedAllList}
          onChange={(value)=>setAllCheckedList(value)}
        >
          {ITEMS.map(item => {
            return (
              <Checkbox key={item} value={item}>
                {item}
              </Checkbox>
            );
          })}
        </Checkbox.Group>
    </div>)
}}
</Playground>
